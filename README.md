# fun-with-turtle
making of square using turtle in python programming
import turtle

shashwat=turtle.Turtle()
shashwat.forward(100)
shashwat.right(90)           # Rotate clockwise by 90 degrees
shashwat.forward(100)
shashwat.right(90)
shashwat.forward(100)
shashwat.right(90)
shashwat.forward(100)
shashwat.right(90)
